@Composable
fun Greeting(name: String, modifier: Modifier = Modifier) {

        Column(horizontalAlignment = Alignment.CenterHorizontally) {
            val langs = remember { mutableStateListOf("Kotlin", "Java", "JavaScript", "Python", "C#", "C++", "Rust")  }
            var count by remember { mutableStateOf(0) }
            LazyColumn( horizontalAlignment = Alignment.CenterHorizontally,
                modifier = Modifier
                    //.padding(16.dp)
                    // Добавление границы: 2.dp, цвет Gray, закругление 8.dp
                    .border(width = 2.dp, color = Color.Gray, shape = RoundedCornerShape(8.dp))

            ){
                items(langs){lang -> Button(onClick = {}) {Text(text = lang, fontSize = 24.sp)} }
            }
            Button(onClick = {
                count++
                langs.add("Элемент $count")
                println(langs)
            }) {
                Text("Добавить элементы")
            }
        }



}
